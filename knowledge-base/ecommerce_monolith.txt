E-Commerce Monolith — Full Source Code
=======================================

This is a monolithic C++ e-commerce application called "ShopCore".
All business logic — authentication, product catalog, order processing,
payment, notifications, and inventory — lives in a single codebase
with tightly coupled classes.

File: shopcore_monolith.hpp
---------------------------

// ============================================================
// DATA MODELS — shared across all modules
// ============================================================

struct User {
    int id;
    std::string username;
    std::string email;
    std::string password_hash;
    std::string role;  // "customer", "admin"
    bool is_active = true;
};

struct Product {
    int id;
    std::string name;
    std::string description;
    double price;
    int stock_quantity;
    std::string category;
    bool is_available = true;
};

struct OrderItem {
    int product_id;
    std::string product_name;
    int quantity;
    double unit_price;
    double subtotal() const { return quantity * unit_price; }
};

struct Order {
    int id;
    int user_id;
    std::vector<OrderItem> items;
    double total_amount;
    std::string status;  // "pending", "paid", "shipped", "delivered", "cancelled"
    std::string created_at;
    std::string payment_id;
};

struct PaymentResult {
    bool success;
    std::string transaction_id;
    std::string error_message;
};

struct Notification {
    int user_id;
    std::string type;   // "email", "sms", "push"
    std::string subject;
    std::string body;
    bool sent = false;
};

// ============================================================
// DATABASE LAYER — single in-memory store for everything
// ============================================================

class MonolithDatabase {
public:
    std::map<int, User> users;
    std::map<int, Product> products;
    std::map<int, Order> orders;
    std::vector<Notification> notification_log;

    std::mutex db_mutex;
    int next_user_id = 1;
    int next_product_id = 1;
    int next_order_id = 1;

    static MonolithDatabase& instance() {
        static MonolithDatabase db;
        return db;
    }

private:
    MonolithDatabase() = default;
};

// ============================================================
// AUTHENTICATION MODULE
// ============================================================

class AuthService {
public:
    static std::string hash_password(const std::string& password) {
        // Simplified hash for demo — production would use bcrypt/argon2
        std::hash<std::string> hasher;
        return std::to_string(hasher(password));
    }

    static std::optional<User> register_user(const std::string& username,
                                              const std::string& email,
                                              const std::string& password) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);

        // Check for duplicate email
        for (const auto& [id, user] : db.users) {
            if (user.email == email) {
                std::cerr << "Registration failed: email already exists\n";
                return std::nullopt;
            }
        }

        User user;
        user.id = db.next_user_id++;
        user.username = username;
        user.email = email;
        user.password_hash = hash_password(password);
        user.role = "customer";
        db.users[user.id] = user;

        // TIGHT COUPLING: Auth directly sends welcome notification
        NotificationService::send_email(user.id, "Welcome!",
            "Welcome to ShopCore, " + username + "!");

        return user;
    }

    static std::optional<User> login(const std::string& email,
                                      const std::string& password) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);

        for (const auto& [id, user] : db.users) {
            if (user.email == email &&
                user.password_hash == hash_password(password) &&
                user.is_active) {
                return user;
            }
        }
        return std::nullopt;
    }

    static bool deactivate_user(int user_id) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);
        if (db.users.count(user_id)) {
            db.users[user_id].is_active = false;
            return true;
        }
        return false;
    }
};

// ============================================================
// NOTIFICATION MODULE — called directly by other modules
// ============================================================

class NotificationService {
public:
    static void send_email(int user_id, const std::string& subject,
                           const std::string& body) {
        auto& db = MonolithDatabase::instance();
        Notification notif;
        notif.user_id = user_id;
        notif.type = "email";
        notif.subject = subject;
        notif.body = body;
        notif.sent = true;
        db.notification_log.push_back(notif);
        std::cout << "[EMAIL] To user " << user_id << ": " << subject << "\n";
    }

    static void send_sms(int user_id, const std::string& message) {
        auto& db = MonolithDatabase::instance();
        Notification notif;
        notif.user_id = user_id;
        notif.type = "sms";
        notif.subject = "SMS";
        notif.body = message;
        notif.sent = true;
        db.notification_log.push_back(notif);
        std::cout << "[SMS] To user " << user_id << ": " << message << "\n";
    }
};

// ============================================================
// INVENTORY MODULE — manages stock, coupled to products
// ============================================================

class InventoryService {
public:
    static bool check_stock(int product_id, int quantity) {
        auto& db = MonolithDatabase::instance();
        if (!db.products.count(product_id)) return false;
        return db.products[product_id].stock_quantity >= quantity;
    }

    static bool reserve_stock(int product_id, int quantity) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);

        if (!db.products.count(product_id)) return false;
        auto& product = db.products[product_id];
        if (product.stock_quantity < quantity) return false;

        product.stock_quantity -= quantity;

        // TIGHT COUPLING: Inventory directly triggers low-stock alerts
        if (product.stock_quantity < 5) {
            NotificationService::send_email(0, "Low Stock Alert",
                "Product '" + product.name + "' has only " +
                std::to_string(product.stock_quantity) + " units left.");
        }
        return true;
    }

    static void restock(int product_id, int quantity) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);
        if (db.products.count(product_id)) {
            db.products[product_id].stock_quantity += quantity;
        }
    }
};

// ============================================================
// PAYMENT MODULE — processes payments, coupled to orders
// ============================================================

class PaymentService {
public:
    static PaymentResult process_payment(int user_id, double amount,
                                          const std::string& method) {
        // Simulated payment processing
        PaymentResult result;
        if (amount <= 0) {
            result.success = false;
            result.error_message = "Invalid amount";
            return result;
        }

        // Simulate gateway call
        result.success = true;
        result.transaction_id = "TXN-" + std::to_string(user_id) + "-" +
                                std::to_string(static_cast<int>(amount * 100));
        return result;
    }

    static bool refund(const std::string& transaction_id, double amount) {
        std::cout << "[PAYMENT] Refund " << amount << " for " << transaction_id << "\n";

        // TIGHT COUPLING: Payment directly notifies user about refund
        // (but doesn't even have the user_id cleanly — has to parse it)
        return true;
    }
};

// ============================================================
// PRODUCT CATALOG MODULE
// ============================================================

class CatalogService {
public:
    static Product add_product(const std::string& name, const std::string& desc,
                                double price, int stock, const std::string& category) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);

        Product p;
        p.id = db.next_product_id++;
        p.name = name;
        p.description = desc;
        p.price = price;
        p.stock_quantity = stock;
        p.category = category;
        db.products[p.id] = p;
        return p;
    }

    static std::vector<Product> search_products(const std::string& keyword) {
        auto& db = MonolithDatabase::instance();
        std::vector<Product> results;
        for (const auto& [id, product] : db.products) {
            if (product.is_available &&
                (product.name.find(keyword) != std::string::npos ||
                 product.description.find(keyword) != std::string::npos)) {
                results.push_back(product);
            }
        }
        return results;
    }

    static std::optional<Product> get_product(int product_id) {
        auto& db = MonolithDatabase::instance();
        if (db.products.count(product_id)) {
            return db.products[product_id];
        }
        return std::nullopt;
    }

    static std::vector<Product> get_by_category(const std::string& category) {
        auto& db = MonolithDatabase::instance();
        std::vector<Product> results;
        for (const auto& [id, product] : db.products) {
            if (product.category == category && product.is_available) {
                results.push_back(product);
            }
        }
        return results;
    }
};

// ============================================================
// ORDER MODULE — the most coupled; touches everything
// ============================================================

class OrderService {
public:
    static std::optional<Order> place_order(int user_id,
                                             const std::vector<std::pair<int, int>>& cart) {
        auto& db = MonolithDatabase::instance();

        // Validate user exists
        if (!db.users.count(user_id) || !db.users[user_id].is_active) {
            std::cerr << "Order failed: invalid user\n";
            return std::nullopt;
        }

        // Build order items, check stock for each
        std::vector<OrderItem> items;
        double total = 0.0;

        for (const auto& [product_id, qty] : cart) {
            auto product = CatalogService::get_product(product_id);
            if (!product) {
                std::cerr << "Order failed: product " << product_id << " not found\n";
                return std::nullopt;
            }
            if (!InventoryService::check_stock(product_id, qty)) {
                std::cerr << "Order failed: insufficient stock for " << product->name << "\n";
                return std::nullopt;
            }

            OrderItem item;
            item.product_id = product_id;
            item.product_name = product->name;
            item.quantity = qty;
            item.unit_price = product->price;
            total += item.subtotal();
            items.push_back(item);
        }

        // Process payment
        auto payment = PaymentService::process_payment(user_id, total, "credit_card");
        if (!payment.success) {
            std::cerr << "Order failed: payment error — " << payment.error_message << "\n";

            // TIGHT COUPLING: Order service sends notification on payment failure
            NotificationService::send_email(user_id, "Payment Failed",
                "Your payment of $" + std::to_string(total) + " could not be processed.");
            return std::nullopt;
        }

        // Reserve inventory
        for (const auto& [product_id, qty] : cart) {
            InventoryService::reserve_stock(product_id, qty);
        }

        // Create order
        std::lock_guard<std::mutex> lock(db.db_mutex);
        Order order;
        order.id = db.next_order_id++;
        order.user_id = user_id;
        order.items = items;
        order.total_amount = total;
        order.status = "paid";
        order.payment_id = payment.transaction_id;
        order.created_at = "2025-06-15T10:30:00Z";
        db.orders[order.id] = order;

        // TIGHT COUPLING: Order service directly sends confirmation email + SMS
        NotificationService::send_email(user_id, "Order Confirmed",
            "Your order #" + std::to_string(order.id) +
            " for $" + std::to_string(total) + " has been placed.");
        NotificationService::send_sms(user_id,
            "Order #" + std::to_string(order.id) + " confirmed! Total: $" +
            std::to_string(total));

        return order;
    }

    static bool cancel_order(int order_id) {
        auto& db = MonolithDatabase::instance();
        std::lock_guard<std::mutex> lock(db.db_mutex);

        if (!db.orders.count(order_id)) return false;
        auto& order = db.orders[order_id];
        if (order.status == "shipped" || order.status == "delivered") return false;

        // Refund payment
        PaymentService::refund(order.payment_id, order.total_amount);

        // Restore inventory
        for (const auto& item : order.items) {
            InventoryService::restock(item.product_id, item.quantity);
        }

        order.status = "cancelled";

        // TIGHT COUPLING: cancellation notification
        NotificationService::send_email(order.user_id, "Order Cancelled",
            "Order #" + std::to_string(order_id) + " has been cancelled. "
            "Refund of $" + std::to_string(order.total_amount) + " initiated.");

        return true;
    }

    static std::vector<Order> get_user_orders(int user_id) {
        auto& db = MonolithDatabase::instance();
        std::vector<Order> user_orders;
        for (const auto& [id, order] : db.orders) {
            if (order.user_id == user_id) {
                user_orders.push_back(order);
            }
        }
        return user_orders;
    }
};
